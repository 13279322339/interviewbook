```
class Value {
	public int i = 15;
}
public class Test10 {

	public static void main(String argv[]) {
		Test10 t = new Test10();
		t.first();
	}

	public void first() {
		int i = 5;
		Value v = new Value();
		v.i = 25;  // v.i =  20
		second(v, 5);
		System.out.println(v.i);
	}

	public void second(Value v, int i) {
		i = 0;
		v.i = 20;
		Value val = new Value();
		v = val;
		System.out.println(v.i + " " + i);
	}
}
```
