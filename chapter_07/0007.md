# 如何保证一个类的非静态方法在多个线程中是同步的?

# 答案

{%ace edit=false, lang='java', theme='monokai'%}
class InstanceMethodThreadSafe {
  private static final Object mutex = new Object();

  public void m1() {
    synchronized (mutex) {
      // do something ...
    }
  }

  public void m2() {
    synchronized (mutex) {
      // do something ...
    }
  }
}
{%endace%}
